64_Count palindrome in range

Given two integers L and R, your task is to count how many numbers in the range [L, R] are palindromes.
A number is called a palindrome if it reads the same forward and backward.

import java.util.*;

public class Main {
    public static boolean isPal(int n) {
        int rev = 0, temp = n;
        while (n > 0) {
            rev = rev * 10 + n % 10;
            n /= 10;
        }
        return rev == temp;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int l = sc.nextInt(), r = sc.nextInt(), count = 0;

        for (int i = l; i <= r; i++) {
            if (isPal(i)) count++;
        }
        System.out.println(count);
    }
}
