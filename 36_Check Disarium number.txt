44.Find all factors of a number

Given a positive integer n, find all the distinct divisors of n


import java.util.*;
public class Main
{
	public static void main(String[] args) {
	    Scanner sc = new Scanner(System.in);
	    int n = sc.nextInt();
	    for(int i=1;i<=n;i++){
	    if(n%i==0){
	        System.out.println(i);
	
	}
	
	    }
	}
}
