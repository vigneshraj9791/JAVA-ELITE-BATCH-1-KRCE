32_Count Armstrongs in range:

Count how many numbers within a given range satisfy the Armstrong condition (sum of digits raised to the power of number of digits equals the number).
Iterate through the range, check each number, and increment the count for every Armstrong number found.

Program :
import java.util.*;
public class Main {
    public static void main(String[] args){
        Scanner sc = new Scanner(System.in);
        int start = sc.nextInt();
        int end = sc.nextInt();
        int count = 0;
        for(int i = start; i <= end; i++){
            int t = i;
            int sum = 0, digits = 0;
            int temp = i;
            while(temp != 0){
                temp /= 10;
                digits++;
            }

            temp = i;
            while(temp != 0){
                int d = temp % 10;
                sum += (int)Math.pow(d, digits);
                temp /= 10;
            }

            if(sum == i){
                count++;
            }
        }

        System.out.println(count);
    }
}
