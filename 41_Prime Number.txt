66) Count primes in range

Given a range L to R, count how many numbers are prime numbers.
A prime number is a number greater than 1 that has exactly two factors: 1 and itself.

import java.util.*;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int l=sc.nextInt(),r=sc.nextInt(),count=0;
        for(int i=l;i<=r;i++){
            if(i<2) continue;
            int flag=1;
            for(int j=2;j*j<=i;j++){
                if(i%j==0){ flag=0; break; }
            }
            if(flag==1) count++;
        }
        System.out.println(count);
    }
}
