12_Remove all 0s from number

Given an integer number N, the task is to remove all zeroes from the given number.

Program:
import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int a=sc.nextInt();
		int rem=0;
		int result=0;
		int digit=1;
		int pow=1;
		while(a>0){
		    rem=a%10;
		    a/=10;
		    if(rem==0){
		        continue;
		    }
		    result+=rem*pow;
		    pow=(int)Math.pow(10,digit++);
		}
		System.out.println(result);
	}
}
