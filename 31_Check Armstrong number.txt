79)Find next palindrome number

Given a number N, find the smallest palindrome number greater than N.

import java.util.*;
public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        n++;

        while(true){
            int temp = n, rev = 0;
            while(temp > 0){
                rev = rev * 10 + temp % 10;
                temp /= 10;
            }
            if(rev == n){
                System.out.println(n);
                break;
            }
            n++;
        }
    }
}
