15_Rotate digits by k positions

Given two integers N and K, the task is to rotate the digits of N by K. If K is a positive integer, left rotate its digits. Otherwise, right rotate its digits.

Program:
import java.util.*;

public class Main {
    public static int countDigits(int n) {
        int count = 0;
        while (n > 0) {
            count++;
            n /= 10;
        }
        return count;
    }

    public static int rotate(int n, int k) {
        int d = countDigits(n);
        k = k % d;

        if (k < 0) {
            k = k + d;
        }

        int pow1 = (int)Math.pow(10, d - k);
        int pow2 = (int)Math.pow(10, k);

        int firstPart = n / pow1;
        int secondPart = n % pow1;

        return (secondPart * pow2) + firstPart;
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int k = sc.nextInt();
        System.out.println(rotate(n, k));
    }
}
